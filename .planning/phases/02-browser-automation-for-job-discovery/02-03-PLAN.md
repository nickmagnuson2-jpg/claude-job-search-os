**`tools/career_scanner/scanner.py`:** Main orchestrator.

```python
import yaml
import sys
import time
from pathlib import Path
from datetime import datetime, timezone

def load_targets(repo_root: Path) -> list[dict]:
    """Load active targets from scan-targets.yaml."""
    config_path = repo_root / "data" / "scan-targets.yaml"
    if not config_path.exists():
        print(f"No scan-targets.yaml found at {config_path}", file=sys.stderr)
        return []
    with open(config_path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f)
    return [c for c in data.get("companies", []) if c.get("active", True)]

def fetch_company_roles(target: dict) -> list[dict]:
    """Fetch roles for a single company using the appropriate parser."""
    ats = target.get("ats", "generic")
    slug = target.get("slug", "")
    name = target.get("name", slug)
    
    if ats == "greenhouse":
        from tools.career_scanner.greenhouse import fetch_greenhouse
        roles = fetch_greenhouse(slug)
    elif ats == "lever":
        from tools.career_scanner.lever import fetch_lever
        roles = fetch_lever(slug)
    elif ats == "ashby":
        from tools.career_scanner.ashby import fetch_ashby
        roles = fetch_ashby(slug)
    elif ats == "generic":
        from tools.career_scanner.generic import fetch_generic
        roles = fetch_generic(target.get("careers_url", ""), name)
    else:
        print(f"Unknown ATS '{ats}' for {name}", file=sys.stderr)
        return []
    
    # Override company name with display name from config
    for r in roles:
        r["company"] = name
    
    # Apply role_filters if specified
    filters = target.get("role_filters", [])
    if filters:
        roles = [r for r in roles if any(
            f.lower() in r["title"].lower() for f in filters
        )]
    
    return roles

def scan_all_targets(repo_root: Path, dry_run: bool = False) -> dict:
    """
    Full pipeline: load targets -> fetch -> score -> dedup -> write inbox.
    Returns summary dict with counts and results.
    """
    from tools.career_scanner.scorer import score_role, load_scoring_context
    from tools.career_scanner.dedup import load_pipeline_entries, filter_duplicates
    
    targets = load_targets(repo_root)
    if not targets:
        return {"total_fetched": 0, "new_roles": 0, "skipped_dupes": 0, "companies_scanned": 0}
    
    scoring_ctx = load_scoring_context(repo_root)
    pipeline_entries = load_pipeline_entries(repo_root)
    
    all_roles = []
    for i, target in enumerate(targets):
        print(f"Scanning {target.get('name', '?')} ({target.get('ats', '?')})...", file=sys.stderr)
        roles = fetch_company_roles(target)
        all_roles.extend(roles)
        # Rate limit: 0.5s between companies (T-02-02 mitigation)
        if i < len(targets) - 1:
            time.sleep(0.5)
    
    # Score all roles
    for role in all_roles:
        role["score"] = score_role(role, scoring_ctx)
    
    # Dedup against pipeline
    new_roles, skipped = filter_duplicates(all_roles, pipeline_entries)
    
    # Sort by score descending (D-05)
    new_roles.sort(key=lambda r: r.get("score", 0), reverse=True)
    
    summary = {
        "total_fetched": len(all_roles),
        "new_roles": len(new_roles),
        "skipped_dupes": skipped,
        "companies_scanned": len(targets),
        "roles": new_roles,
    }
    
    if new_roles and not dry_run:
        write_inbox(repo_root, new_roles)
    
    return summary

def format_inbox_entry(roles: list[dict]) -> str:
    """Format scan results as inbox entry text."""
    now_str = datetime.now(timezone.utc).strftime("%Y-%m-%d %H:%M UTC")
    lines = [
        f"## Career Scan Results",
        f"",
        f"**Scanned:** {now_str}",
        f"**New roles found:** {len(roles)}",
        f"",
    ]
    for role in roles:
        score = role.get("score", 0)
        loc = role.get("location", "")
        lines.append(
            f"- **[{score}/10]** {role['title']} at {role['company']}"
            + (f" ({loc})" if loc else "")
            + f" - [View]({role['url']})"
        )
    lines.append(f"")
    lines.append(f"*Source: /scan-companies | {now_str}*")
    lines.append(f"")
    return "\n".join(lines)

def write_inbox(repo_root: Path, roles: list[dict]):
    """Prepend scan results to data/inbox.md following granola_auto_debrief.py pattern."""
    inbox_path = repo_root / "data" / "inbox.md"
    entry = format_inbox_entry(roles)
    
    existing = ""
    if inbox_path.exists():
        with open(inbox_path, "r", encoding="utf-8") as f:
            existing = f.read()
    
    # Find insertion point after header (# Inbox + comment block)
    lines = existing.split("\n") if existing else []
    insert_at = 0
    for i, line in enumerate(lines):
        if line.startswith("# ") or line.startswith("<!--") or line.startswith("-->") or line.strip() == "":
            insert_at = i + 1
        else:
            break
    
    new_content = "\n".join(lines[:insert_at]) + "\n" + entry + "\n".join(lines[insert_at:])
    with open(inbox_path, "w", encoding="utf-8") as f:
        f.write(new_content)
    
    print(f"Wrote {len(roles)} roles to {inbox_path}", file=sys.stderr)
```

**`tools/career_scanner/cli.py`:** CLI entrypoint.

```python
#!/usr/bin/env python3
"""CLI for career scanner. Run with: PYTHONIOENCODING=utf-8 python tools/career_scanner/cli.py"""
import argparse
import json
import sys
from pathlib import Path

def main():
    p = argparse.ArgumentParser(description="Scan career pages for matching roles")
    p.add_argument("--dry-run", action="store_true", help="Fetch and score but don't write to inbox")
    p.add_argument("--repo-root", default=None, help="Repository root path")
    args = p.parse_args()
    
    repo_root = Path(args.repo_root).resolve() if args.repo_root else Path(__file__).parent.parent.parent
    
    from tools.career_scanner.scanner import scan_all_targets
    result = scan_all_targets(repo_root, dry_run=args.dry_run)
    
    # Print summary to stdout as JSON
    output = {k: v for k, v in result.items() if k != "roles"}
    output["top_roles"] = [
        {"title": r["title"], "company": r["company"], "score": r.get("score", 0), "url": r["url"]}
        for r in result.get("roles", [])[:10]
    ]
    print(json.dumps(output, indent=2, ensure_ascii=False))

if __name__ == "__main__":
    main()
```
